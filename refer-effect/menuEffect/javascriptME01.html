// 1. 버튼을 클릭하면 클릭한 버튼한테 active 추가
// 2. 버튼을 클릭하지 않은 다른 버튼은 active 제거
// 3. 버튼 인덱스 설정
// 4. 1번째 버튼은 1번째 컨텐츠 연결(버튼 인덱스 별 해당 컨텐츠 연결)
// 5. 모든 컨텐츠를 안보이게 설정
// 6. 클릭한 버튼의 해당 컨텐츠만 보이게 설정

const tabBtn = document.querySelectorAll(".tab-btn ul li");
const tabCont = document.querySelectorAll(".tab-cont > div");

tabBtn.forEach((element, index) => {
    element.addEventListener("click", function(){
        
        tabBtn.forEach(el => {
            el.classList.remove("active");
        });
        element.classList.add("active");

        tabCont.forEach(el => {
            el.style.display = "none";
        })
        tabCont[index].style.display = "block";
    });
})
