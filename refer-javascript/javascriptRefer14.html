07. 테이블 만들기1 

{
    document.write("*** 07. 테이블 만들기1 ***");

    let num = 1;
    let table = "<table>";

    for( let i=1; i<=10; i++ ){
        table += "<tr>";
        for( let j=1; j<=10; j++ ){
            table += "<td>" + num + "</td>";
            num++;    
            }
        table += "</tr>";
    }

    table += "</table>";

    document.write(table);
}
